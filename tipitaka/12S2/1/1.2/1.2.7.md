# 𑀅𑀘𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑀼𑀢𑁆𑀢 (1.2.7 Acelakassapasutta)

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢 (Nidānasaṃyutta)

* 𑀆𑀳𑀸𑀭𑀯𑀕𑁆𑀕 (Āhāravagga)

* 𑀅𑀘𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑀼𑀢𑁆𑀢 (Acelakassapasutta)

## Summary

Acelakassapasutta records a conversation between the Buddha and the naked Kassapa, who asks the Buddha about the nature of dukkha. The Buddha rejects Kassapa's questions about dukkha, and instead teaches the middle way between Eternalism and Nihilism using dependent origination. Kassapa is impressed by the Buddha's teaching, take the three refuges and asks to be ordained in the Buddha's presence. As he had formerly belonged to another sect (of naked ascetics), he chose to undergo probation for 4 years rather than the required 4 months. A short time after ordination, Kassapa became arahataṃ (awakened).

## Translation

(17.)

96\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀅𑀘𑁂𑀮𑁄 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀅𑀘𑁂𑀮𑁄 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀓𑀜𑁆𑀘𑀺𑀤𑁂𑀯 𑀤𑁂𑀲𑀁, 𑀲𑀘𑁂 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀑𑀓𑀸𑀲𑀁 𑀓𑀭𑁄𑀢𑀺 𑀧𑀜𑁆𑀳𑀲𑁆𑀲 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄 𑀢𑀸𑀯, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀜𑁆𑀳𑀲𑁆𑀲; 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀸”𑀢𑀺𑁇

> Thus heard by me: At one time the Bhagavā was dwelling in Rājagaha, Veḷuvana (bamboo forest), Kalandakanivāpa (a park where squirrels are fed). Then the Bhagavā, having dressed in the morning, having taken his bowl and robe, entered Rājagaha for alms. The naked Kassapa saw the Bhagavā coming from afar. Having seen him, he approached the Bhagavā; having approached, he exchanged greetings with the Bhagavā. Having exchanged greetings, the naked Kassapa stood to one side. The naked Kassapa, standing to one side, said to the Bhagavā:
>
> > “I would ask the Bhagavā Gotama about some matter, if the Bhagavā Gotama has enough time to answer the question.”
>
> > “It is not the right time for the question, Kassapa; we have entered amongst the houses (in the village).”

97\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀘𑁂𑀮𑁄 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀓𑀜𑁆𑀘𑀺𑀤𑁂𑀯 𑀤𑁂𑀲𑀁, 𑀲𑀘𑁂 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀑𑀓𑀸𑀲𑀁 𑀓𑀭𑁄𑀢𑀺 𑀧𑀜𑁆𑀳𑀲𑁆𑀲 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄 𑀢𑀸𑀯, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀜𑁆𑀳𑀲𑁆𑀲; 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀸”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀘𑁂𑀮𑁄 𑀓𑀲𑁆𑀲𑀧𑁄…𑀧𑁂…  𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀸𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀘𑁂𑀮𑁄 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦 𑀔𑁄 𑀧𑀦 𑀫𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀩𑀳𑀼𑀤𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀓𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀧𑀼𑀘𑁆𑀙, 𑀓𑀲𑁆𑀲𑀧, 𑀬𑀤𑀸𑀓𑀗𑁆𑀔𑀲𑀻”𑀢𑀺𑁇

> For a second time the naked ascetic Kassapa said to the Bhagavā:
>
> > “I would ask the Bhagavā Gotama about some matter, if the Bhagavā Gotama has enough time to answer the question.”
>
> > “It is not the right time for the question, Kassapa; we have entered amongst the houses.”
>
> For a third time the naked ascetic Kassapa …  entered amongst the houses."
>
> When this was said, the naked ascetic Kassapa said to the Bhagavā:
>
> > “I don't wish to ask the Bhagavā Gotama many questions.”
>
> > “Ask, Kassapa, whatever you wish.”

98\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, ‘𑀲𑀬𑀁𑀓𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺? ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀧𑀭𑀁𑀓𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺? ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺? ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺? ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺? ‘𑀦 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺𑁇 ‘𑀢𑁂𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 ‘𑀦 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀁; 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’”𑀢𑀺𑁇

> > “Bho Gotama, is dukkha 'sayaṃkataṃ' (self-made)?”
>
> “Don't say that, Kassapa,” the Bhagavā said.
>
> > “Bho Gotama, is dukkha 'paraṃkataṃ' (other-made)?”
>
> “Don't say that, Kassapa,” the Bhagavā said.
>
> > “Bho Gotama, is dukkha 'sayaṃkatañca paraṃkatañca' (both self-made and other-made)?”
>
> “Don't say that, Kassapa,” the Bhagavā said.
>
> > If so, Bho Gotama, is dukkha 'asayaṃkāraṃ aparaṅkāraṃ adhiccasamuppannaṃ' (not done by oneself, not done by another, occurring spontaneously)?”
>
> “Don't say that, Kassapa,” the Bhagavā said.
>
> > “What, Bho Gotama, is there no dukkha?”
>
> “Not so, Kassapa; there is dukkha, Kassapa.”
>
> > “Then does Bhagavā Gotama not know or see dukkha?”
>
> “Certainly not, Kassapa. I do know dukkha, Kassapa; and I do see dukkha, Kassapa.”

99\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, ‘𑀲𑀬𑀁𑀓𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀧𑀭𑀁𑀓𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀦 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀢𑁂𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀦 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀦 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀁; 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀼 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀤𑁂𑀲𑁂𑀢𑀼 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺𑁇

> > "To this exact question, 'Bho Gotama, is dukkha self made?', you replied 'Don't say that, Kassapa'."
> >
> > "To this exact question, 'Bho Gotama, is dukkha other-made?', you replied 'Don't say that, Kassapa'."
> >
> > "To this exact question, 'Bho Gotama, is dukkha both self-made and other-made?', you replied 'Don't say that, Kassapa'."
> >
> > "To this exact question, 'If so, Bho Gotama, is dukkha not done by oneself, not done by another, occurring spontaneously?', you replied 'Don't say that, Kassapa'."
> >
> > "To this exact question, 'What, Bho Gotama, is there no dukkha', you replied 'Not so, Kassapa; there is dukkha, Kassapa.'."
> >
> > "To this exact question, 'Then does Bhagavā Gotama not know or see dukkha??', you replied 'Certainly not, Kassapa. I do know dukkha, Kassapa; and I do see dukkha, Kassapa.'."
> >
> > "May the Bhagavā describe dukkha to me, Bhante, may the Bhagavā teach dukkha to me, Bhante."

100\. “‘𑀲𑁄 𑀓𑀭𑁄𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀆𑀤𑀺𑀢𑁄 𑀲𑀢𑁄 ‘𑀲𑀬𑀁𑀓𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀤𑀁 𑀲𑀲𑁆𑀲𑀢𑀁 𑀏𑀢𑀁 𑀧𑀭𑁂𑀢𑀺𑁇 ‘𑀅𑀜𑁆𑀜𑁄 𑀓𑀭𑁄𑀢𑀺 𑀅𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀯𑁂𑀤𑀦𑀸𑀪𑀺𑀢𑀼𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀢𑁄 ‘𑀧𑀭𑀁𑀓𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀤𑀁 𑀉𑀘𑁆𑀙𑁂𑀤𑀁 𑀏𑀢𑀁 𑀧𑀭𑁂𑀢𑀺𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻’”𑀢𑀺𑁇

> "So karoti so paṭisaṃvedayatī." (one does, one personally experiences.) - initially others who are Eternalists say "Dukkha is self-made."
>
> "Añño karoti añño paṭisaṃvedayati" (one does, another personally experiences.) - when afflicted with pain and suffering, others who are Nihilists say "Dukkha is other-made."
>
> Not approaching either of these extremes, the Tathāgata teaches the Dhamma majjhena (by the middle):
>
> > "With avijjā as cause, saṅkhārā; with saṅkhārā as cause, viññāṇaṁ…  thus is the origin of this entirety of dukkha. With the complete cessation of avijjā, the cessation of saṅkhārā; with the cessation of saṅkhārā, the cessation of viññāṇaṁ…  thus is the cessation of this entirety of dukkha.”

101\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀘𑁂𑀮𑁄 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬…𑀧𑁂…  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇

> When this was said, the naked Kassapa said to the Bhagavā:
>
> > “Excellent, Bhante, excellent, Bhante. Just like, Bhante, what is turned upside down is turned right side up … one who has vision sees forms'; so too the Bhagavā has explained the Dhamma in many ways. Bhante, I go to the Bhagavā for saraṇaṃ (refuge), to the Dhamma, and to the bhikkhusaṅgha (community of bhikkhus). May I receive pabbajjaṃ (the going forth) in the Bhagavā’s presence; may I receive upasampadaṃ (the higher ordination)?”

102\. “𑀬𑁄 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀼𑀩𑁆𑀩𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁, 𑀲𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀭𑀺𑀯𑀲𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀆𑀭𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀅𑀧𑀺 𑀘 𑀫𑀬𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀯𑀺𑀤𑀺𑀢𑀸”𑀢𑀺𑁇

> “Whoever, Kassapa, that is aññatitthiyapubbo (formerly belonging to another sect), wishes to go forth and take up this dhammavinaye (doctrine and discipline), he lives under probation for four months. At the end of the four months, the bhikkhus āraddhacittā (satisfied with his accomplishments) confirms his going forth and ordains him into bhikkhubhāvāya (bhikkhu status). Furthermore, I have known puggalavemattatā (the personality of the person).”

103\. “𑀲𑀘𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀼𑀩𑁆𑀩𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀭𑀺𑀯𑀲𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀆𑀭𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀅𑀳𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀧𑀭𑀺𑀯𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀆𑀭𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀼 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀸𑀬𑀸”𑀢𑀺𑁇

> > “If, Bhante, someone formerly belonging to another sect - who wishes to go forth and take up this dhammavinaye - has to live under probation for four months, and if at the end of the four months, the bhikkhus satisfied with his accomplishments confirms his going forth and ordains him into bhikkhubhāvāya, then I will live under probation for four years, at the end of four years, the bhikkhus satisfied with my accomplishments can confirm my going forth and ordains me into bhikkhubhāvāya.”

104\. 𑀅𑀮𑀢𑁆𑀣 𑀔𑁄 𑀅𑀘𑁂𑀮𑁄 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀢𑁆𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀅𑀘𑀺𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇

> Then the naked Kassapa received the going forth in the Bhagavā’s presence; and (in due course) he received the higher ordination. Not long after his ordination, the Āyasmā Kassapa, living for a short time solitary, secluded, diligent, ardent, and resolute - there is no higher purpose than a young man from a good family residing in a dwelling going forth into homelessness (renunciating) - with these dhamme (teachings or truths), by sayaṃ (his own effort), sacchikatvā (having personal experienced) and abhiññā (with direct knowledge), he attained and remained in brahmacariyapariyosānaṃ (the perfection of the holy life).
>
> He completely knew:
>
> > “Khīṇā jāti, vusitaṃ brahmacariyaṃ, kataṃ karaṇīyaṃ, nāparaṃ itthattāyā” (My (purpose for) being is no more, my spiritual path finished, the duty completed, nothing further in this lifetime.”)
>
> And certainly the Āyasmā Kassapa became arahataṃ (awakened).

## Commentary

This sutta has a tetralemma that illustrates that "self" and "dukkha" are disjointed concepts that should not be linked together. The middle way that the Buddha teaches is the dependent origination, between Eternalism and Nihilism.

This sutta is interesting in that it may imply the Buddha can still experience dukkha after awakening, but the Buddha is able to "unconstruct" experienced dukkha, and hence extinguish and eliminate it. This is supported by the Buddha clearly experiencing dukkha in his latter years, falling sick, troubled by events, becoming old and eventually dying, so he is clearly subject to jarāmaraṇaṃ and hence dukkha. Therefore the elimination of dukkha is a psychological elimation, the removal of the mental results of stimuli and physical processes, not the avoidance of them. This also implies cessation or extinguishment are ongoing activities and not once-off.

It is also interesting that it discloses one of the Buddha's rules regarding ordination - religious practitioners from other sects must undergo four months probation before they can be fully ordained.
